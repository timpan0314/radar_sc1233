178 static rs_ret_t rs_start_body(rs_handle_t handle, uint8_t resume) {
179  struct rs_handle *lh = (struct rs_handle *) handle;
180 
181  rs_ret_t ret = RS_OK;
182 
183  RET_CHECK(handle != NULL, RS_EINV);
184 
185  ret = rs24g_mutex_lock(lh->mutex);
186 
187  if(ret == RS_OK) {
188  if(lh->state != RS_STATE_READY) {
189  ret = RS_ESTATE;
190  }
191 
192  if(ret == RS_OK) {
193  ret = rs_ctl_dev_term_trig_clear(lh->dev, RS_TERM_OR);
194  }
195  if(ret == RS_OK) {
196  ret = rs_ctl_dev_term_trig_clear(lh->dev, RS_TERM_DETOUT);
197  }
198 
199  if(ret == RS_OK) {
200  ret = rs_ctl_op_SRST(lh->dev, 1);
201  }
202  rs_usleep(5 * 1000);
203 
204  if(ret == RS_OK) {
205  uint32_t value;
206  if (resume) {
207  value = 0x00000100;
208  } else {
209  value = 0x000009F0;
210  }
211  ret = rs_ctl_write_reg(lh->dev, RS_CTL_REG_SELDIG_06, value);
212  }
213  if(ret == RS_OK) {
214  ret = rs_ctl_write_reg(lh->dev, RS_CTL_REG_SELDIG_06, 0x00000000);
215  }
216 
217  if(ret == RS_OK) {
218  uint8_t status = 0;
219  ret = rs_ctl_op_RDSR(lh->dev, &status);
220  if(ret == RS_OK) {
221  if((status & RS_CTL_OP_STATUS_SEQ_BUSY) != 0x00) {
222  ret = RS_EDEV;
223  }
224  }
225  }
226 
227  // can't write registers in deep sleep state on I2C
228  if(lh->sensor_ctl_type == RS_SENSOR_CTL_TIMER) {
229  if(ret == RS_OK) {
230  ret = rs_ctl_write_reg(lh->dev, RS_CTL_REG_CLKGEN,
231  (RS_CTL_REG_CLKGEN_SEQ_ENABLE |
232  RS_CTL_REG_CLKGEN_TIMER_ENABLE));
233  }
234  if(ret == RS_OK) {
235  ret = rs_ctl_op_DSLEEP(lh->dev);
236  }
237 
238  if(ret == RS_OK) {
239  ret = rs_ctl_op_ENATM(lh->dev);
240  }
241  if(ret == RS_OK) {
242  ret = rs_ctl_op_RUNTM(lh->dev);
243  }
244  } else if(lh->sensor_ctl_type == RS_SENSOR_CTL_CPU) {
245  if(ret == RS_OK) {
246  ret = rs_ctl_op_WRSR(lh->dev, RS_CTL_MODE_CONT_START);
247  }
248  }
249 
250  if(ret == RS_OK) {
251  lh->state = RS_STATE_SENSING;
252  }
253 
254  if(ret == RS_OK) {
255  ret = rs24g_mutex_unlock(lh->mutex);
256  } else {
257  rs24g_mutex_unlock(lh->mutex);
258  }
259  }
260 
261  return ret;
262 }

276 static rs_ret_t rs_stop_wait(struct rs_handle *lh) {
277  rs_ret_t ret = RS_OK;
278 
279  RET_CHECK(lh != NULL, RS_EINV);
280 
281  if(lh->sensor_ctl_type == RS_SENSOR_CTL_TIMER) {
282  /* wait interval time */
283 #if defined(RS_CPUIF_USE_I2C)
284  if(ret == RS_OK) {
285  ret = rs_ctl_op_HLDDT(lh->dev, 1, 1);
286  }
287 #endif
288  if(ret == RS_OK) {
289  uint32_t rdata;
290  ret = rs_ctl_read_reg(lh->dev, RS_CTL_REG_CPUIF_4, &rdata);
291  if (ret == RS_OK) {
292  rs_usleep((rdata / 0x40 + 1) * 1000);
293  }
294  }
295 #if defined(RS_CPUIF_USE_I2C)
296  if(ret == RS_OK) {
297  ret = rs_ctl_op_UPDDT(lh->dev, 1);
298  }
299 #endif
300  } else if(lh->sensor_ctl_type == RS_SENSOR_CTL_CPU) {
301  /* wait not SEQ_BUSY */
302  if(ret == RS_OK) {
303  uint8_t status = 0;
304  time_t start, current;
305  time(&start);
306  for(;;) {
307  ret = rs_ctl_op_RDSR(lh->dev, &status);
308  if((status & RS_CTL_OP_STATUS_SEQ_BUSY) == 0x00) {
309  break;
310  }
311  time(&current);
312  if(current > start + RS_CONTINUOUS_STOP_WAIT_SEC) {
313  ret = RS_EDEV;
314  break;
315  }
316  rs_usleep(100);
317  }
318  }
319  }
320  return ret;
321 }
322 #endif
