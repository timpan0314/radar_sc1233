45 static rs_ret_t rs_ctl_op_write_region(rs_ctl_dev_t dev, uint32_t address, const uint8_t *data, rs_size_t size)
46 {
47 #if defined(RS_LOW_MEMORY)
48  uint8_t wdata[4];
49  wdata[0] = RS_CTL_OP_WRITE_DATA;
50  wdata[1] = (address >> 16) & 0xff;
51  wdata[2] = (address >> 8) & 0xff;
52  wdata[3] = (address >> 0) & 0xff;
53  return rs_ctl_dev_write_low_memory(dev, wdata, sizeof(wdata), data, size);
54 #else
55  rs_ret_t ret;
56  rs_size_t wsize = 4 + size;
57  uint8_t *wdata = (uint8_t *)malloc(wsize);
58  if (wdata == NULL)
59  {
60  return RS_ENOMEM;
61  }
62  wdata[0] = RS_CTL_OP_WRITE_DATA;
63  wdata[1] = (address >> 16) & 0xff;
64  wdata[2] = (address >> 8) & 0xff;
65  wdata[3] = (address >> 0) & 0xff;
66  memcpy(&wdata[4], data, size);
67  ret = rs_ctl_dev_write(dev, wdata, wsize);
68  free(wdata);
69  return ret;
70 #endif
71 }
