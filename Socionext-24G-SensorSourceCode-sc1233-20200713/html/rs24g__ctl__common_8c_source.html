179 {

184

186  data->distance[1] = (int16_t) ((regs_distance[ 0] >> 0) & 0xffff);
187  data->distance[2] = (int16_t) ((regs_distance[ 1] >> 16) & 0xffff);
188  data->distance[3] = (int16_t) ((regs_distance[ 1] >> 0) & 0xffff);
189  data->distance[4] = (int16_t) ((regs_distance[ 2] >> 0) & 0xffff);
190 
191  for(i=0; i<5; i++) {
192  if(regs_peak_level[i] == 0) {
193  data->peak_level[i] = RS_PEAK_LEVEL_INVALID;
194  } else {
195  data->peak_level[i] = conv_power(regs_peak_level[i]);
196  }
197 
198  data->peak_real[i] = (int16_t) ((regs_complex[i] >> 16) & 0xffff);
199  data->peak_imag[i] = (int16_t) ((regs_complex[i] >> 0) & 0xffff);
200  }
201 
202  RET_ORG(rs_ret_t, distance_range(data, peak_level_lower), RS_OK)
203 
204  return RS_OK;
