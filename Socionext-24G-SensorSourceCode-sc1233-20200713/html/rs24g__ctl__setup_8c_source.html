68 static rs_ret_t rs_setup_seq(rs_handle_t handle)
69 {
70  struct rs_handle *lh = (struct rs_handle *) handle;
71  uint16_t expected;
72  uint32_t crc;
73  uint16_t crc_seqpgm;
74  uint16_t crc_fftwin;
75 #if defined(RS_WRITE_MEM_LIMIT)
76  rs_size_t size;
77  uint32_t addr;
78  const uint8_t *data;
79 #endif
80 
81  /* register parameters of code */
82  {
83  size_t i;
84 
85  for (i=0; i<lh->code_ref.regs.num; i++) {
86  RS_CALL(rs_ctl_write_reg(lh->dev, lh->code_ref.regs.addr[i], lh->code_ref.regs.data[i]));
87  }
88  }
89  /* wait until stabilized, as the setting for crystal may be changed at this register update */
90  rs_usleep(5 * 1000);
91  RS_CALL(rs_ctl_cmd_disable_seq(lh->dev));
92  /* write Sequencer code */
93 #if defined(RS_WRITE_MEM_LIMIT)
94  size = lh->code_ref.seqc.size;
95  addr = RS_CTL_TOPADDR_SEQ;
96  data = (const uint8_t *)(lh->code_ref.seqc.data);
97  while (size > 0)
98  {
99  rs_size_t unitsize = size < RS_WRITE_MEM_LIMIT ? size : RS_WRITE_MEM_LIMIT;
100  RS_CALL(rs_ctl_write_mem(lh->dev, addr, data, unitsize));
101  /* check CRC */
102  expected = get_expected_crc(addr, data, unitsize);
103  RS_CALL(rs_ctl_read_reg(lh->dev, RS_CTL_REG_CPUIF_MON_2, &crc));
104  crc_seqpgm = (crc >> 16) & 0xFFFF;
105  RS_ASSERT(crc_seqpgm == expected, RS_ELOAD);
106  /* increment */
107  addr += unitsize / 16;
108  data += unitsize;
109  size -= unitsize;
110  }
111 #else
112  RS_CALL(rs_ctl_write_mem(lh->dev, RS_CTL_TOPADDR_SEQ, (const uint8_t *)(lh->code_ref.seqc.data), lh->code_ref.seqc.size));
113  /* check CRC of Sequencer code */
114  expected = get_expected_crc(RS_CTL_TOPADDR_SEQ, (const uint8_t *)(lh->code_ref.seqc.data), lh->code_ref.seqc.size);
115  RS_CALL(rs_ctl_read_reg(lh->dev, RS_CTL_REG_CPUIF_MON_2, &crc));
116  crc_seqpgm = (crc >> 16) & 0xFFFF;
117  RS_ASSERT(crc_seqpgm == expected, RS_ELOAD);
118 #endif
119  /* write Window function */
120 #if defined(RS_WRITE_MEM_LIMIT)
121  size = lh->code_ref.winf.size;
122  addr = RS_CTL_TOPADDR_FFT;
123  data = (const uint8_t *)(lh->code_ref.winf.data);
124  while (size > 0)
125  {
126  rs_size_t unitsize = size < RS_WRITE_MEM_LIMIT ? size : RS_WRITE_MEM_LIMIT;
127  RS_CALL(rs_ctl_write_mem(lh->dev, addr, data, unitsize));
128  /* check CRC */
129  expected = get_expected_crc(addr, data, unitsize);
130  RS_CALL(rs_ctl_read_reg(lh->dev, RS_CTL_REG_CPUIF_MON_2, &crc));
131  crc_fftwin = (crc >> 0) & 0xFFFF;
132  RS_ASSERT(crc_fftwin == expected, RS_ELOAD);
133  /* increment */
134  addr += unitsize / 2;
135  data += unitsize;
136  size -= unitsize;
137  }
138 #else
139  RS_CALL(rs_ctl_write_mem(lh->dev, RS_CTL_TOPADDR_FFT, (const uint8_t *)(lh->code_ref.winf.data), lh->code_ref.winf.size));
140  /* check CRC of Window function */
141  expected = get_expected_crc(RS_CTL_TOPADDR_FFT, (const uint8_t *)(lh->code_ref.winf.data), lh->code_ref.winf.size);
142  RS_CALL(rs_ctl_read_reg(lh->dev, RS_CTL_REG_CPUIF_MON_2, &crc));
143  crc_fftwin = (crc >> 0) & 0xFFFF;
144  RS_ASSERT(crc_fftwin == expected, RS_ELOAD);
145 #endif
146  RS_CALL(rs_ctl_cmd_enable_seq(lh->dev));
147 
148  return RS_OK;
149 }

154 RS_IF rs_ret_t rs_load_seqcode(rs_handle_t handle, rs_mode_t mode)
155 {
156  struct rs_handle *lh = (struct rs_handle *) handle;
157 
158  rs_ret_t ret = RS_OK;
159

161 
162  ret = rs24g_mutex_lock(lh->mutex);
163 
164  if(ret == RS_OK) {
165  if(lh->state != RS_STATE_SHUTDOWN) {
166  ret = RS_ESTATE;

184  ret = rs24g_mutex_unlock(lh->mutex);
