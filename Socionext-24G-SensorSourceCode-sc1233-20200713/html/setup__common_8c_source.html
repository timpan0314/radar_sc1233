135 rs_ret_t rs_setup_range(rs_handle_t handle, uint16_t addr, uint32_t upper, uint32_t lower, int32_t conv_offset)
136 {
137  const uint32_t fft_point = 512;
138  const double a = RS_C * point_per_chirp(handle, fft_point) / (2 * effective_chirp_bandwidth(fft_point) * fft_point);
139 
140  rs24gsetup_dp("conv_offset:%d", conv_offset);
141  rs24gsetup_dp("upper:%d", upper);
142  rs24gsetup_dp("lower:%d", lower);
143 
144  uint32_t up = (uint32_t) ceil(512 / 2 - (upper - conv_offset) / a);
145  uint32_t lo = (uint32_t) floor(512 / 2 - (lower - conv_offset) / a);
146  rs24gsetup_dp("up:%d lo:%d", up, lo);
147 
148  {
149  uint32_t range[16];
150  uint32_t i;
151  for(i=0; i<sizeof(range)/sizeof(*range); i++) {
152  range[i] = 0;
153  }
154  for(i=0; i<512; i++) {
155  if((lo >= i) && (i >= up)) {
156  range[15 - i / 32] |= 0x00000001 << (i % 32);
157  }
158  }
159  for(i=0; i<sizeof(range)/sizeof(*range); i++) {
160  RET_ORG(rs_ret_t, read_modify_write_reg(handle, addr + i, 0xffffffff, range[i]), RS_OK);
161  }
162  }
163 
164  return RS_OK;
165 }
