47 RS_IF rs_ret_t rs_ctl_cmd_boot(rs_ctl_dev_t dev)
48 {
49  // CE/NRST setting
50  RS_CALL(rs_ctl_dev_term_set(dev, RS_TERM_CE, RS_TRUE));
51  rs_usleep(2 * 1000);
52  RS_CALL(rs_ctl_dev_term_set(dev, RS_TERM_NRST, RS_TRUE));
53 
54  RS_CALL(rs_ctl_op_SRST(dev, 1));
55  rs_usleep(5 * 1000);
56 
57  // read efuse
58  RS_CALL(rs_ctl_write_reg(dev, RS_CTL_REG_RD_START_EFUSE, 0x00000001));
59  rs_usleep(100);
60  RS_CALL(rs_ctl_write_reg(dev, RS_CTL_REG_CLKGEN, 0x001FFC44));
61  {
62  uint32_t rvalue = 0;
63  RS_CALL(rs_ctl_read_reg(dev, RS_CTL_REG_EFUSE37, &rvalue));
64  if ((rvalue & 0x000000FF) != 0) {
65  RS_CALL(rs_ctl_write_reg(dev, RS_CTL_REG_SELSET_F_4, 0x00000000));
66  }
67  }
68 
69  RS_CALL(rs_ctl_op_DSLEEP(dev));
70 
71  return RS_OK;
72 }

190 static rs_ret_t read_registers(rs_ctl_dev_t dev, const struct rs_ctl_sensor_data_set *set, struct rs_ctl_sensor_data *data)
191 {
192  if(set->reg.enable) {
193  rs_size_t i = 0;
194  data->reg.num = set->reg.num;
195  RS_ASSERT((data->reg.data = (uint32_t *) calloc(set->reg.num, sizeof(uint32_t))) != NULL, RS_ENOMEM);
196 #if !defined(RS_CPUIF_USE_I2C)
197  RS_CALL(rs_ctl_op_HLDDT(dev, 1, 0));
198 #endif
199  for(i=0; i<set->reg.num; i++) {
200  RS_CALL(rs_ctl_read_reg(dev, set->reg.addr[i], &(data->reg.data[i])));
201  }
202 #if !defined(RS_CPUIF_USE_I2C)
203  RS_CALL(rs_ctl_op_UPDDT(dev, 1));
204 #endif
205  }
206  return RS_OK;
207 }

213 static rs_ret_t read_fifo(rs_ctl_dev_t dev, const struct rs_ctl_sensor_data_set *set, struct rs_ctl_sensor_data *data)
214 {
215  if(set->sram.enable) {
216  rs_size_t wordsize_sram = 0;
217  wordsize_sram += set->sram.exist_header_header == RS_TRUE ? 1 : 0;
218  wordsize_sram += set->sram.exist_header_status == RS_TRUE ? 1 : 0;
219  wordsize_sram += set->sram.wordsize_body;
220  wordsize_sram += 1;
221  data->sram.size = wordsize_sram * set->sram.bytesize_word;
222 
223  RS_ASSERT((data->sram.data = (uint8_t *) calloc(wordsize_sram, set->sram.bytesize_word)) != NULL, RS_ENOMEM);
224  RS_CALL(rs_ctl_read_mem(dev, RS_CTL_TOPADDR_FIFO, data->sram.data, data->sram.size));
225  }
226  return RS_OK;
227 }
